<script>
    const totalSlots = 50;
    let bookedSlots = [];
    let selectedSlot = null;

    function initializeSlots() {
        const container = document.getElementById('parkingSlots');
        container.innerHTML = '';

        for (let i = 1; i <= totalSlots; i++) {
            const slot = document.createElement('div');
            slot.className = bookedSlots.includes(i) ? 'slot booked' : 'slot available';
            slot.textContent = "P" + i;

            slot.onclick = () => selectSlot(i, slot);
            container.appendChild(slot);
        }

        updateStats();
    }

    function selectSlot(num, element) {
        if (element.classList.contains('booked')) return;

        selectedSlot = num;
        document.getElementById('slotNumber').value = num;

        document.querySelectorAll('.slot').forEach(s => s.classList.remove('selected'));
        element.classList.add('selected');
    }

    function updateStats() {
        document.getElementById('availableSlots').textContent = totalSlots - bookedSlots.length;
        document.getElementById('bookedSlots').textContent = bookedSlots.length;
    }

    document.getElementById('bookingForm').addEventListener('submit', function(e) {
        e.preventDefault();

        const vehicle = document.getElementById('vehicleNumber').value.trim();
        const name = document.getElementById('ownerName').value.trim();
        const phone = document.getElementById('phoneNumber').value.trim();
        const duration = document.getElementById('duration').value;

        const vehicleRegex = /^[A-Z]{2}[0-9]{2}[A-Z]{2}[0-9]{4}$/;
        const nameRegex = /^[A-Za-z ]{3,}$/;
        const phoneRegex = /^[0-9]{10}$/;

        if (!vehicleRegex.test(vehicle)) {
            alert("Invalid vehicle number (DL01AB1234)");
            return;
        }

        if (!nameRegex.test(name)) {
            alert("Name must contain only letters");
            return;
        }

        if (!phoneRegex.test(phone)) {
            alert("Phone must be 10 digits");
            return;
        }

        if (duration < 1 || duration > 24) {
            alert("Duration must be 1–24 hours");
            return;
        }

        if (!selectedSlot) {
            alert("Select a slot first");
            return;
        }

        bookedSlots.push(selectedSlot);

        alert("Slot P" + selectedSlot + " booked successfully!");

        selectedSlot = null;
        this.reset();
        initializeSlots();
    });

    initializeSlots();
</script>
