    <script>
        const totalSlots = 50;
        let bookedSlots = [];

        function initializeSlots() {
            const slotsContainer = document.getElementById('parkingSlots');
            slotsContainer.innerHTML = '';
            for (let i = 1; i <= totalSlots; i++) {
                const slot = document.createElement('div');
                slot.className = bookedSlots.includes(i) ? 'slot booked' : 'slot available';
                slot.textContent = `P${i}`;
                slot.onclick = () => selectSlot(i, slot);
                slotsContainer.appendChild(slot);
            }
            updateStats();
        }

        function selectSlot(slotNumber, slotElement) {
            if (slotElement.classList.contains('available')) {
                document.getElementById('vehicleNumber').focus();
                document.getElementById('slotNumber') = slotNumber;
            }
        }

        function updateStats() {
            const available = totalSlots - bookedSlots.length;
            document.getElementById('availableSlots').textContent = available;
            document.getElementById('bookedSlots').textContent = bookedSlots.length;
        }

        document.getElementById('bookingForm').addEventListener('submit', function(e) {
            e.preventDefault();
            alert('Booking submitted! Please complete payment to confirm.');
            this.reset();
        });

        initializeSlots();
    </script>
